##### 流程六：系统类
##### uml源码如下
```
@startuml

package "图书管理系统类包" #DDDDDD {

	class 图书管理员 {

		+维护图书()
		+归还图书()
		+借出图书()
		+维护读者信息()
	}
	class 读者 {

		+查询借阅图书信息()
		+预订图书()
		+取消预订图书()
	}
	class 游客 {
		+查询图书信息()
	}

	class 图书 {
	 	-书名
	 	-ISBN号
	 	-总量
	 	-库存
	 	-价格
	 	-出版社
	 	-简介
	 	-作者
	}

	class 借书记录 {
	 	-读者姓名
	 	-图书ISBN号
	 	-借出时间
	 	-归还期限
	}
	class 还书记录 {
    	 	-读者姓名
    	 	-图书ISBN号
    	 	-还书时间
    	 	-借书时长
    	}

	 note "<color:white>省略Getter和Setter方法。</color>" as note1

	图书 .. note1
	note1 .. 借书记录
	游客 <|-- 读者 : 注册登录
	读者 "1" *-- "N" 借书记录 : 借书
	读者 "1" *-- "N" 还书记录 : 还书
	借书记录 "1" o-- "1" 图书 : 一一对应
	借书记录 "N" -- "1" 图书管理员 : 登记和借出
	图书管理员 "N" -- "N" 图书 : 维护图书
	图书管理员 "N" -- "N" 读者 : 维护读者信息
	还书记录 "N" -- "1" 图书管理员 : 登记和归还
    还书记录 "1" o-- "1" 图书 : 一一对应

}


@enduml
```
